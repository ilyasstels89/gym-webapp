    <!-- ЭКРАН ВВОДА -->

    <div id="input-screen">

        <div class="stat-box">

            <h3 id="set-counter">Подход №1</h3>

            <p style="opacity: 0.7">Время тренировки: <span id="total-time">00:00</span></p>

        </div>


        <input type="text" id="exercise" placeholder="Название упражнения (Жим)">

        <div style="display: flex; gap: 10px;">

            <input type="tel" id="weight" placeholder="Вес (кг)">

            <input type="tel" id="reps" placeholder="Повторы">

        </div>


        <button class="btn btn-primary" onclick="logSet()">✅ Сделал и отдыхать</button>

        <button class="btn btn-danger" onclick="finishWorkout()">🏁 Закончить тренировку</button>

    </div>


    <!-- ЭКРАН ТАЙМЕРА -->

    <div id="timer-screen">

        <h2>Отдых</h2>

        <div id="timer-display">01:30</div>

        <br>

        <button class="btn btn-primary" onclick="stopRest()">💪 К следующему подходу</button>

        <p style="margin-top: 20px; opacity: 0.5;">+10 сек / -10 сек</p>

    </div>


    <script>

        let tg = window.Telegram.WebApp;

        tg.expand(); // На весь экран


        let setNumber = 1;

        let workoutData = [];

        let workoutStartTime = Date.now();

        let restInterval;

        let totalTimeInterval;


        // Общий таймер тренировки

        totalTimeInterval = setInterval(() => {

            let diff = Math.floor((Date.now() - workoutStartTime) / 1000);

            let m = Math.floor(diff / 60).toString().padStart(2, '0');

            let s = (diff % 60).toString().padStart(2, '0');

            document.getElementById('total-time').innerText = `${m}:${s}`;

        }, 1000);


        function logSet() {

            let exercise = document.getElementById('exercise').value;

            let weight = document.getElementById('weight').value;

            let reps = document.getElementById('reps').value;


            if(!exercise || !weight) {

                tg.showAlert("Заполни название и вес!");

                return;

            }


            // Сохраняем подход

            workoutData.push({

                set: setNumber,

                exercise: exercise,

                weight: weight,

                reps: reps

            });


            setNumber++;

            document.getElementById('set-counter').innerText = `Подход №${setNumber}`;

            

            // Запускаем отдых

            startRest(90); // 90 секунд по умолчанию

        }


        function startRest(seconds) {

            document.getElementById('input-screen').style.display = 'none';

            document.getElementById('timer-screen').style.display = 'block';

            

            let timeLeft = seconds;

            updateTimerDisplay(timeLeft);


            restInterval = setInterval(() => {

                timeLeft--;

                updateTimerDisplay(timeLeft);

                if(timeLeft <= 0) {

                    tg.HapticFeedback.notificationOccurred('success'); // Вибрация

                    stopRest();

                }

            }, 1000);

        }


        function updateTimerDisplay(sec) {

            let m = Math.floor(sec / 60).toString().padStart(2, '0');

            let s = (sec % 60).toString().padStart(2, '0');

            document.getElementById('timer-display').innerText = `${m}:${s}`;

        }


        function stopRest() {

            clearInterval(restInterval);

            document.getElementById('timer-screen').style.display = 'none';

            document.getElementById('input-screen').style.display = 'block';

        }


        function finishWorkout() {

            if (workoutData.length === 0) {

                tg.close();

                return;

            }

            // Отправляем данные боту

            tg.sendData(JSON.stringify(workoutData));

        }
